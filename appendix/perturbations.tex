\chapter{宇宙論的摂動}\label{app:perturbations}

宇宙は完全に一様等方ではない。これは星や銀河が存在していることからも自明である。そこで、一様等方宇宙からの摂動を考える。
\begin{equation}
 \phi(t, \bm{x}) = \bar{\phi}(t) + \delta\phi(t, \bm{x}), \quad g_{\mu\nu}(t, \bm{x}) = \bar{g}_{\mu\nu}(t) + \delta g_{\mu\nu}(t, \bm{x})
\end{equation}
ここで、$\bar{\phi}(t)$と$\bar{g}_{\mu\nu}(t)$は一様成分を表し、$\delta \phi(t, \bm{x})$と$\delta{g_{\mu\nu}}(t, \bm{x})$はその摂動を表す。この計量における線素$\mathrm{d}s^{2}$は、
\begin{equation}
\begin{aligned}
 \mathrm{d}s^{2} &= g_{\mu\nu}\mathrm{d}x^{\mu}\mathrm{d}x^{\nu} \\
 &= -(1 + 2\Phi)\mathrm{d}t^{2} + 2a B_{i}\mathrm{d}x^{i}\mathrm{d}t + a^{2}[(1 - \Psi)\delta_{ij} + E_{ij})]\mathrm{d}x^{i}\mathrm{d}x^{j}
 \end{aligned}
\end{equation}
で表される。ここで、$B_{i}$、$E_{ij}$は、
\begin{equation}
 \begin{aligned}
  B_{i}  := \partial_{i}B - S_{i}, \quad (\partial^{i}S_{i} &= 0), \\
  E_{ij} := 2\partial_{ij}E + 2\partial_{(i}F_{j)} + h_{ij}, \quad (\partial^{i} F_{i} &= 0, \ h^{i}{}_{i} = \partial^{i}h_{ij} = 0)
 \end{aligned}
\end{equation}
である。実空間における、スカラー・ベクトル・テンソル摂動は、
それぞれ分解できることが知られており、独立に考えられる。
また、ベクトル摂動$S_{i}$、$F_{i}$は、
宇宙の膨張にしたがって減衰する成分であるとしてここでは扱わない。

テンソル摂動は、座標変換
\begin{equation}
 \begin{aligned}
  t     &\rightarrow t + \alpha \\
  x^{i} &\rightarrow x^{i} + \delta^{ij}\beta_{,j}
 \end{aligned}
\end{equation}
に対して不変（ゲージ不変）だが、スカラー摂動は次のような変換を受ける。
\begin{align}
 \Phi &\rightarrow \Phi - \dot{\alpha} \\
 B    &\rightarrow B + a^{-1}\alpha - a\dot{\beta} \\
 E    &\rightarrow E - \beta \\
 \Psi &\rightarrow \Psi + H\alpha
\end{align}
また、スカラー場の摂動$\delta \phi$は
\begin{equation}
 \delta \phi \rightarrow \delta \phi - \dot{\phi}\delta t
\end{equation}
と変換される。

\subsubsection{スカラー摂動}

スカラー場$\phi$のスローロール・インフレーションにおける作用$S$を考える。
\begin{equation}
 S = \frac{1}{2}\int \mathrm{d}^{4}x\sqrt{-g}
  \left[
   R - (\Delta \phi)^{2} -2V(\phi)
  \right]
  \label{eq:S_slow-roll}
\end{equation}
ここで、$c=1$、$\hbar = 1$、$M_{\mathrm{pl}}^{-2}:=8\pi G=1$となるような単位系を採用し、$R$はリッチ・スカラーである。また、次のようなゲージを選択する。
\begin{equation}
 \delta \phi = 0, \quad g_{ij} = a^{2}\left[(1 - 2\mathcal{R})\delta_{ij} + h_{ij}\right], \quad \partial^{i}h_{ij} = h^{i}{}_{i} = 0
\end{equation}
このゲージにおいて、インフラトン場は摂動を受けない。またスカラー成分の自由度は$\mathcal{R}(t, \bm{x})$で表される。

式(\ref{eq:S_slow-roll})の2次摂動は、
\begin{equation}
 S_{(2)} = \frac{1}{2}\int \mathrm{d}^{4}x a^{3}\frac{\dot{\phi}^{2}}{H^{2}}\left[\dot{\mathcal{R}}^{2} - a^{-2}(\partial_{i}\mathcal{R})^{2}\right]
  \label{eq:S_2}
\end{equation}
となる。ここで、
\begin{equation}
 v := z\mathrm{R}, \quad z^{2} := a^{2}\frac{\dot{\phi}^{2}}{H^{2}}
\end{equation}
と定義して、式(\ref{eq:S_2})を共形時間で表せば、
\begin{equation}
 S_{(2)} = \frac{1}{2}\int \mathrm{d}\tau\mathrm{d}^{3}x \left[(v')^{2} + (\partial_{i} v)^{2} + \frac{z''}{z}v^{2}\right]
\end{equation}
となる。ここで、$(\cdots)':=\partial_{\tau}(\cdots)$である。

スカラー場$v$のフーリエ変換を
\begin{equation}
 v(\tau, \bm{x}) := \int \frac{\mathrm{d}^{3}k}{(2\pi)^{3}}v_{\bm{k}}(\tau)\mathrm{e}^{\mathrm{i}\bm{k}\cdot \bm{x}}
\end{equation}
で定義すれば、次の方程式（ムハノフ--佐々木方程式）が導かれる。
\begin{equation}
 v_{k}'' + \left(k^{2} - \frac{z''}{z}\right)v_{k} = 0
  \label{eq:mukhanov_eq}
\end{equation}
ここで、式(\ref{eq:mukhanov_eq})は、$k:=|\bm{k}|$にしか依らない。

スカラー場$v$を、調和振動子の量子化を類推して、量子化する。
\begin{equation}
\begin{aligned}
 v &\rightarrow \hat{v} = \int \frac{\mathrm{d}\bm{k}^{3}}{(2\pi)^{3}}
 \left[
 v_{k}(\tau)\hat{a}_{\bm{k}}\mathrm{e}^{\mathrm{i}\bm{k}\cdot \bm{x}}
 + v_{k}^{*}(\tau)\hat{a}_{\bm{k}}^{\dagger}\mathrm{e}^{-\mathrm{i}\bm{k}\cdot \bm{x}}
 \right], \\
 v_{\bm{k}} &\rightarrow \hat{v}_{\bm{k}} = v_{k}(\tau)\hat{a}_{\bm k}
 + v_{-k}^{*}(\tau)\hat{a}_{-\bm{k}}^{\dagger}
\end{aligned}
\end{equation}
ここで、$\hat{a}_{\bm{-k}}^{\dagger}$、$\hat{a}_{\bm k}$は、生成、消滅演算子で、次のような正準交換関係を満たす。
\begin{equation}
 [\hat{a}_{\bm k}, \hat{a}_{\bm k'}^{\dagger}]
  = (2\pi)^{3}\delta(\bm{k} - \bm{k}')
\end{equation}
ただし、モード関数のロンスキアン$\left<v_{k}, v_{k}\right>$は次のように規格化されているとする。
\begin{equation}
 \left<v_{k}, v_{k}\right>
  := \frac{\mathrm{i}}{\hbar}(v_{k}^{*}v_{k}' - v_{k}^{*}{'}v_{k}) = 1
  \label{eq:wronskian}
\end{equation}
式(\ref{eq:wronskian})は、式\ref{eq:mukhanov_eq}の解に対して、ひとつの境界条件を与える。ふたつめの境界条件は、真空の選択によるモード関数の固定である。これにより、方程式に解が与えられる。

そこでゆらぎの真空状態を次のように選ぶ。
\begin{equation}
 \left.\hat{a}_{\bm k}|0\right> = 0
\end{equation}





スカラー摂動は、一様な超曲面上$\xi$の曲率摂動により与えられる\cite{PhysRevD.28.679}。
\begin{equation}
 -\xi := \Psi + \frac{H}{\dot{\bar{\rho}}}\delta \rho
  \label{eq:xi}
\end{equation}
式(\ref{eq:xi})は、スローロール・インフレーションの場合、
\begin{equation}
 -\xi \approx \Psi + \frac{H}{\bar{\dot{\phi}}}\delta \phi =: \mathcal{R}
  \label{eq:xi'}
\end{equation}
と近似できる。スカラーゆらぎのパワースペクトルは、
\begin{equation}
 \left<\mathcal{R}_{\bm{k}}\mathcal{R}_{\bm{k}'}\right>
  = (2\pi)^{3}\delta(\bm{k} + \bm{k'})P_{\mathcal{R}}(k),
  \quad \Delta_{\mathrm s}^{2}
  := \Delta_{\mathrm{\mathcal R}}^{2}
  = \frac{k^{3}}{2\pi^{2}}P_{\mathcal R}(k)
  \label{eq:ps}
\end{equation}
となる。ここで、$\left<\cdots\right>$は、ゆらぎのアンサンブル平均である。パワースペクトルのスケール依存性を考えるとき、次のような指標が用いられる。
\begin{equation}
 n_{\mathrm s} - 1 := \frac{\mathrm{d}\ln \Delta_{\mathrm{s}}^{2}}{\mathrm{d}\ln k}, \quad \alpha_{\mathrm s} := \frac{\mathrm{d}n_{\mathrm s}}{\mathrm{d}\ln k}
  \label{eq:n_s}
\end{equation}
ここで、スケール不変であるとき、$n_{\mathrm s}=1$である。
spatially-flatゲージを考えると、式(\ref{eq:xi'})において$\Psi=0$となり、$\mathcal{R}$とインフレーションのスカラー場の関係は、
\begin{equation}
 \left< \mathcal{R}_{\bm{k}} \mathcal{R}_{\bm{k'}} \right>
  = \left(H \over \dot{\phi}\right)^{2} \left< \delta \phi_{\bm{k}} \delta \phi_{\bm{k'}}\right>
  \label{eq:R_and_phi}
\end{equation}
となり、スローロール・インフレーションにおける、スカラー場（$m_{\phi}\ll H$）の量子ゆらぎは、ハッブル・パラメーターを使って、
\begin{equation}
 \left< \delta\phi_{\bm{k}} \delta\phi_{\bm{k'}} \right>
  = (2\pi)^{3}\delta(\bm{k} + \bm{k'})\frac{H^{2}}{2k^{3}}
  \label{eq:fluc_phi}
\end{equation}
となる。したがって、式(\ref{eq:ps})、(\ref{eq:R_and_phi})、(\ref{eq:fluc_phi})より、スカラー場がつくるゆらぎのパワースペクトルは次のようになる。
\begin{equation}
 \Delta_{\mathrm s}^{2}(k) = \left.\left(\frac{H}{2\pi}\right)^{2}{H^{2}\over 2k^{3}}\right|_{k=aH}
\end{equation}

\subsubsection{テンソル摂動}

テンソル摂動には2つのモード$h_{ij}$（$h := h^{+},\ h^{\times}$）が存在する。このテンソルゆらぎのパワースペクトルは、
\begin{equation}
 \left< h_{\bm{k}} h_{\bm{k'}} \right> = (2\pi)^{3}\delta(\bm{k} + \bm{k'})P_{h}(k)
  , \quad
  \Delta_{h}^{2} = \frac{k^{3}}{2\pi^{2}}P_{h}(k)
\end{equation}
となる。2つのモードを合わせたパワースペクトルは、
\begin{equation}
 \Delta_{\mathrm t}^{2} := 2\Delta_{h}^{2} = \left.\frac{2}{\pi^{2}}\left(H\over M_{\mathrm{pl}}\right)^{2}\right|_{k=aH}
  \label{eq:ps_tensor}
\end{equation}
で与えられる。スケール依存性の指標は、スカラー摂動のときと同様にして、次のように定義される。
\begin{equation}
 n_{\mathrm t} := \frac{\mathrm{d}\ln \Delta_{\mathrm t}^{2}}{\mathrm{d}\ln k}
\end{equation}
